52    public static FileFilter HTML_FILE_FILTER = new FileFilter() {
53      @Override
54      public boolean accept(File pathname) {
55        return pathname.isFile() && pathname.getPath().matches(".*\\.htm[l]?\\Z");
56      }
57    };
58    
59    public static FileFilter DIR_FILTER = new FileFilter() {
60      @Override
61      public boolean accept(File pathname) {
62        return pathname.isDirectory();
63      }
64    };
65    
66    public static FileFilter NOT_INF_DIR_FILTER = new FileFilter() {
67      @Override
68      public boolean accept(File pathname) {
69        return pathname.isDirectory() && !pathname.getName().endsWith("-INF");
70      }
71    };

73    public static Comparator<File> FILE_NAME_COMPARATOR = new Comparator<File>() {
74      @Override
75      public int compare(File file1, File file2) {
76        return file1.getName().compareTo(file2.getName());
77      }
78    };

80    /**
81     * A template class that scans the content directory, starting at the root, and
82     * calls scan(File) for every file that matches the provided content filter.
83     * 
84     * @author Christian Trimble
85     */
86    public static abstract class ContentScanTemplate
87    {
88      private FileFilter contentFilter;
89  
90      public ContentScanTemplate(FileFilter contentFilter) {
91        this.contentFilter = contentFilter;
92      }
93      
94      public void scan( File contentRoot ) throws Exception {
95        // create the frontier and add the content root.
96        LinkedList<File> frontier = new LinkedList<File>();
97        
98        // scan the content root dir for html files.
99        for( File htmlFile : contentRoot.listFiles(contentFilter)) {
100         handleFile(htmlFile);
101       }
102       
103       // add the non "-INF" directories, in a predictable order.
104       frontier.subList(0, 0).addAll(Arrays.asList(sort(contentRoot.listFiles(NOT_INF_DIR_FILTER), FILE_NAME_COMPARATOR)));
105       
106       while( !frontier.isEmpty() ) {
107         File dir = frontier.removeFirst();
108         
109         // scan the html files in the directory.
110         for( File htmlFile : dir.listFiles(contentFilter)) {
111           handleFile(htmlFile);
112         }
113    
114         // add the directories, in a predictable order.
115         frontier.subList(0, 0).addAll(Arrays.asList(sort(dir.listFiles(DIR_FILTER), FILE_NAME_COMPARATOR)));
116       }
117     }
118     
119     /**
120      * An call to Arrays.sort(array, comparator) that returns the array argument after the sort.
121      * 
122      * @param array the array to sort.
123      * @param comparator the comparator to sort with.
124      * @return the array argument.
125      */
126     private static <T> T[] sort( T[] array, Comparator<T> comparator ) {
127       Arrays.sort(array, comparator);
128       return array;
129     }
130     
131     public abstract void handleFile( File file )
132       throws Exception;
133   }

141   private final ReadLock readLock = locker.readLock();
142   private final WriteLock writeLock = locker.writeLock();

145   @Override
146   public synchronized void processFromDirectory(String metaDir) throws Exception {
147     SearchHolder newStagedSearcher = new SearchHolder();
148     indexDir = new File(metaDir, "lucene-index");
149     dataDir = new File(metaDir).getParentFile();
150     newStagedSearcher.directory = new NIOFSDirectory(indexDir);
151     IndexWriter iwriter = null;
152     try {
153       iwriter = new IndexWriter(newStagedSearcher.directory, new IndexWriterConfig(Version.LUCENE_36, analyzer).setRAMBufferSizeMB(5));
154       iwriter.deleteAll();
155       writeIndex(iwriter, dataDir);
156     }
157     finally {
158       IOUtils.closeQuietly(iwriter);
159     }
160     newStagedSearcher.indexReader = IndexReader.open(newStagedSearcher.directory);
161     stagedSearch = newStagedSearcher;
162   }
163   
164   void writeIndex( final IndexWriter indexWriter, File contentDir ) throws Exception {
165     new ContentScanTemplate(HTML_FILE_FILTER) {
166       @Override
167       public void handleFile(File file) throws Exception {
168         Jerry jerry = Jerry.jerry(FileUtils.readFileToString(file, "UTF-8"));
169         String title = jerry.$("html > head > title").text();
170         String textContent = jerry.$("html > body").text();
171 
172         Document doc = new Document();
173         doc.add(new Field("title", title, Field.Store.YES, Field.Index.ANALYZED));
174         doc.add(new Field("content", textContent, Field.Store.YES, Field.Index.ANALYZED));
175         doc.add(new Field("path", file.getPath().replaceFirst(dataDir.getPath(), ""), Field.Store.YES, Field.Index.NOT_ANALYZED));
176         indexWriter.addDocument(doc);
177       }
178     }.scan(contentDir); 
179     
180   }

182   @Override
183   public synchronized void makeLive() {
184     writeLock.lock();
185     if( this.stagedSearch != null && this.stagedSearch.directory != null && this.stagedSearch.indexReader != null ) {
186       SearchHolder oldLive = liveSearch;
187       liveSearch = stagedSearch;
188       IOUtils.closeQuietly(oldLive);
189       stagedSearch = null;
190     }
191     writeLock.unlock();
192   }
193   
194   public void finalize() {
195     IOUtils.closeQuietly(liveSearch);
196     IOUtils.closeQuietly(stagedSearch);
197  }

199   @Override
200   public IndexSearcher startSearch() {
201     readLock.lock();
202     if(this.liveSearch != null) {
203       if(this.liveSearch.indexSearcher == null) {
204         IndexSearcher searcher = new IndexSearcher(this.liveSearch.indexReader);
205         this.liveSearch.indexSearcher = searcher;
206       }
207       return this.liveSearch.indexSearcher;
208     }
209     return null;
210   }
211 
212   @Override
213   public void endSearch() {
214     readLock.unlock();
215   }
