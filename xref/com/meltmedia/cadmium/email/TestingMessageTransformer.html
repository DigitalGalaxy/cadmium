108   public MimeMessage transform( MimeMessage message )
109     throws MessagingException
110   {
111     MimeBodyPart sentToBodyPart = newSentToBodyPart( message );
112     MimeBodyPart originalBodyPart = newOriginalBodyPart( message );
113 
114     // create a new multipart content for this message.
115     MimeMultipart multipart = new MimeMultipart(EmailUtil.SUBTYPE_MIXED);
116 
117     // add the parts to the body.
118     multipart.addBodyPart(originalBodyPart);
119     multipart.addBodyPart(sentToBodyPart);
120 
121     // get the new values for all of the headers.
122     InternetAddress newFromAddress = newFromAddress(message);
123     InternetAddress[] newToAddresses = newToAddresses(message);
124     InternetAddress[] newCcAddresses = newCcAddresses(message);
125     InternetAddress[] newBccAddresses = newBccAddresses(message);
126     String newSubject = newSubject(message);
127 
128     // update the message.
129     message.setFrom(newFromAddress);
130     message.setRecipients(Message.RecipientType.TO, newToAddresses);
131     message.setRecipients(Message.RecipientType.CC, newCcAddresses);
132     message.setRecipients(Message.RecipientType.BCC, newBccAddresses);
133     message.setSubject(newSubject);
134     message.setContent(multipart);
135 
136     // save the message.
137     message.saveChanges();
138 
139     if( getLogProperty() ) {
140       try {
141       ByteArrayOutputStream out = new ByteArrayOutputStream();
142       message.writeTo(out);
143       log.info("Email Message Sent:\n{}", out.toString());
144       }
145       catch( IOException ioe ) {
146         throw new MessagingException("Exception thrown while writing message to log.", ioe);
147       }
148     }
149     if( !getSendProperty() ) {
150       message = null;
151     }
152 
153     // return the message.
154     return message;
155   }

187   public MimeBodyPart newSentToBodyPart( MimeMessage message )
188     throws MessagingException
189   {
190     // get information about the original message.
191     Address[] originalFromRecipient = message.getFrom();
192     Address[] originalToRecipient = message.getRecipients(Message.RecipientType.TO);
193     Address[] originalCcRecipient = message.getRecipients(Message.RecipientType.CC);
194     Address[] originalBccRecipient = message.getRecipients(Message.RecipientType.BCC);
195     String originalSubject = message.getSubject();
196 
197     // create the html for the string buffer.
198     StringBuffer html = new StringBuffer();
199     html.append("<html><body><table style=\"width: 100%;\"><tr><td>header</td><td>value</td></tr>");
200     html.append("<tr><td>Subject:</td><td>").append(escape(originalSubject)).append("</td></tr>");
201 
202     // iterate over the addresses.
203     if( originalFromRecipient != null ) {
204       for( int i = 0; i < originalFromRecipient.length; i++ ) {
205         html.append("<tr><td>FROM:</td><td>").append(escape(originalFromRecipient[i])).append("</td></tr>");
206       }
207     }
208     if( originalToRecipient != null ) {
209       for( int i = 0; i < originalToRecipient.length; i++ ) {
210         html.append("<tr><td>TO:</td><td>").append(escape(originalToRecipient[i])).append("</td></tr>");
211       }
212     }
213     if( originalCcRecipient != null ) {
214       for( int i = 0; i < originalCcRecipient.length; i++ ) {
215         html.append("<tr><td>CC:</td><td>").append(escape(originalCcRecipient[i])).append("</td></tr>");
216       }
217     }
218     if( originalBccRecipient != null ) {
219       for( int i = 0; i < originalBccRecipient.length; i++ ) {
220         html.append("<tr><td>BCC:</td><td>").append(escape(originalBccRecipient[i])).append("</td></tr>");
221       }
222     }
223     html.append("</table></body></html>");
224 
225     MimeBodyPart sentToBodyPart = new MimeBodyPart();
226     sentToBodyPart.setContent(html.toString(), "text/html");
227     sentToBodyPart.setHeader("Content-ID", "original-addresses");
228     sentToBodyPart.setDisposition("inline");
229 
230     return sentToBodyPart;
231   }

233   /**
234    * Executes toString() on the given object and returns the string properly escaped for use in 
235    * the original message information section of an email. The string will be properly HTML 
236    * escaped and will be replaced with the empty string if null. 
237    */
238   protected String escape(Object o) {
239     return escapeHtml(defaultToString(o));
240   }
241   
242   private static String defaultToString(Object o) {
243     return o == null? "" : o.toString() == null? "" : o.toString(); 
244   }

246   public MimeBodyPart newOriginalBodyPart( MimeMessage message )
247     throws MessagingException
248   {
249     // get the data handler for this message.
250     DataHandler handler = message.getDataHandler();
251 
252     // add the original body to the new body part.
253     MimeBodyPart originalPart = new MimeBodyPart();
254     originalPart.setDataHandler(handler);
255     originalPart.setHeader("Content-ID", "original-message");
256 
257     return originalPart;
