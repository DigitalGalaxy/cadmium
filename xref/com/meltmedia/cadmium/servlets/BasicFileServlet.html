94    protected void setBasePath(String basePath) throws ServletException {
95      if(basePath == null) {
96        throw new ServletException("Please set the base path in init paramater \"basePath\".");
97      } else {
98        File contentDir = new File(basePath);
99        if(!contentDir.exists()) {
100         throw new ServletException("The basePath \""+basePath+"\" does not exist on the file system.");
101       } else if(!contentDir.isDirectory()) {
102         throw new ServletException("The basePath \""+basePath+"\" exists and is not a directory.");
103       } else if(!contentDir.canRead()) {
104         throw new ServletException("The basePath\""+basePath+"\" cannot be read.");
105       }
106       this.contentDir = contentDir;
107     }
108   }

126   public void processRequest( FileRequestContext context )
127     throws ServletException, IOException {
128     
129     // Find the file to serve in the file system.  This may redirect for welcome files or send 404 responses.
130     if(locateFileToServe(context)) return;
131     
132     // Handle any conditional headers that may be present.
133     if(handleConditions(context)) return;
134     
135     // Sets the content type header.
136     resolveContentType(context);
137     
138     // Sets compress if Accept-Encoding allows for gzip or identity
139     if(checkAccepts(context)) return;
140     
141     try {
142       if(context.file != null) {
143         context.response.setHeader(CONTENT_DISPOSITION_HEADER, "inline;filename=\"" + context.file.getName() + "\"");
144       }
145       context.response.setHeader(ACCEPT_RANGES_HEADER, "bytes");
146       if(context.eTag != null) {
147         context.response.setHeader(ETAG_HEADER, context.eTag);
148       }
149       context.response.setDateHeader(LAST_MODIFIED_HEADER, lastUpdated);
150       if(!context.ranges.isEmpty()) {
151         context.response.setStatus(HttpServletResponse.SC_PARTIAL_CONTENT);
152         String rangeBoundary = RANGE_BOUNDARY + UUID.randomUUID().toString();
153         if(context.ranges.size() > 1) {
154           context.response.setContentType("multipart/byteranges; boundary=" + rangeBoundary);
155           context.response.setStatus(HttpServletResponse.SC_PARTIAL_CONTENT);
156           
157           if(context.sendEntity) {
158             context.in = new FileInputStream(context.file);
159             context.out = context.response.getOutputStream();
160             ServletOutputStream sout = (ServletOutputStream)context.out;
161             for(Range r : context.ranges) {
162               sout.println();
163               sout.println("--"+rangeBoundary);
164               sout.println("Content-Type: " + context.contentType);
165               sout.println("Context-Range: bytes " + r.start + "-" + r.end + "/" + context.file.length());
166               
167               copyPartialContent(context.in, context.out, r);
168             }
169             
170             sout.println();
171             sout.println("--"+rangeBoundary+"--");
172           }
173         } else {
174           Range r = context.ranges.get(0);
175           context.response.setContentType(context.contentType);
176           Long rangeLength = calculateRangeLength(context, r);
177           context.response.setHeader(CONTENT_RANGE_HEADER, "bytes " + r.start + "-" + r.end
178               + "/" + context.file.length());
179           if(context.sendEntity) {
180             context.in = new FileInputStream(context.file);
181             context.out = context.response.getOutputStream();
182             
183             context.response.setHeader(CONTENT_LENGTH_HEADER, rangeLength.toString());
184 
185             copyPartialContent(context.in, context.out, r);
186           }
187         }
188       } else {
189         context.response.setStatus(HttpServletResponse.SC_OK);
190         context.response.setContentType(context.contentType);
191 
192         if( context.sendEntity ) {
193           context.response.setHeader(CONTENT_RANGE_HEADER, "bytes 0-" + context.file.length() + "/" + context.file.length());
194           context.in = new FileInputStream(context.file);
195           context.out = context.response.getOutputStream();
196           if( context.compress ) {
197           
198             context.response.setHeader(CONTENT_ENCODING_HEADER, "gzip");
199             context.out = new GZIPOutputStream(context.out);
200           
201           } else context.response.setHeader(CONTENT_LENGTH_HEADER, new Long(context.file.length()).toString());
202 
203           IOUtils.copy(context.in, context.out);
204         }
205       }
206     }
207     finally {
208       IOUtils.closeQuietly(context.in);
209       IOUtils.closeQuietly(context.out);
210     } 
211   }

226    * Calculates the length of a given range.
227    * 
228    * @param context
229    * @param range
230    * @return
231    */
232   public static Long calculateRangeLength(FileRequestContext context, Range range) {
233     if(range.start == -1) range.start = 0;
234     if(range.end == -1) range.end = context.file.length() - 1;
235     range.length = range.end - range.start + 1;
236     return range.length;
237   }

239   /**
240    * Checks the accepts headers and makes sure that we can fulfill the request.
241    * @param context 
242    * @return
243    * @throws IOException 
244    */
245   private boolean checkAccepts(FileRequestContext context) throws IOException {
246     if (!canAccept(context.request.getHeader(ACCEPT_HEADER), false, context.contentType)) {
247       notAcceptable(context);
248       return true;
249     }
250     
251     if (!(canAccept(context.request.getHeader(ACCEPT_ENCODING_HEADER), false, "identity") || canAccept(context.request.getHeader(ACCEPT_ENCODING_HEADER), false, "gzip"))) {
252       notAcceptable(context);
253       return true;
254     }
255     
256     if (context.request.getHeader(ACCEPT_ENCODING_HEADER) != null && canAccept(context.request.getHeader(ACCEPT_ENCODING_HEADER), true, "gzip") && shouldGzip(context.contentType)) {
257       context.compress = true;
258     }
259     return false;
260   }

262   /**
263    * <p>Checks for inclusion in the gzipList field. Allows basic wild cards.</p>
264    * <p>An empty or null field defaults to gzipping of all content types.</p> 
265    * @param contentType
266    * @return
267    */
268   public boolean shouldGzip(String contentType) {
269     if(gzipList == null || gzipList.isEmpty()) {
270       return true;
271     } else {
272       return gzipList.contains(contentType) || gzipList.contains(contentType.replaceAll("/[^/]+\\Z", "/*")) || gzipList.contains(contentType.replaceAll("\\A[^/]+/", "*/"));
273     }

277   /**
278    * Locates the file to serve.  Returns true if locating the file caused the request to be handled.
279    * @param context
280    * @return
281    * @throws IOException
282    */
283   public boolean locateFileToServe( FileRequestContext context ) throws IOException {
284     context.file = new File( contentDir, context.path);
285     
286     // if the path is not on the file system, send a 404.
287     if( !context.file.exists() ) {
288       context.response.sendError(HttpServletResponse.SC_NOT_FOUND);
289       return true;    
290     }
291     
292     // redirect welcome files if needed.
293     if( handleWelcomeRedirect(context) ) return true;
294     
295     // if the requested file is a directory, try to find the welcome file.
296     if( context.file.isDirectory() ) {
297       context.file = new File(context.file, "index.html");
298     }
299     
300     // if the file does not exist, then terminate with a 404.
301     if( !context.file.exists() ) {
302       context.response.sendError(HttpServletResponse.SC_NOT_FOUND);
303       return true;
304     }
305     
306     return false;
307   }

309   public boolean handleConditions( FileRequestContext context ) throws IOException {
310     // check the conditions
311     context.eTag = context.path+"_"+lastUpdated;
312     
313     context.ifMatch = context.request.getHeader(IF_MATCH_HEADER);
314     if( context.ifMatch != null && !validateStrong(context.ifMatch, context.eTag) ) {
315       context.response.sendError(HttpServletResponse.SC_PRECONDITION_FAILED);
316       return true;
317     }
318     
319     context.range = context.request.getHeader(RANGE_HEADER);
320     
321     try {
322       context.inRangeDate = context.request.getDateHeader(IF_RANGE_HEADER);
323       if(context.inRangeDate != -1 && context.inRangeDate >= lastUpdated) {
324         if(!parseRanges(context)){
325           invalidRanges(context);
326           return true;
327         }
328       }
329     } catch(IllegalArgumentException iae) {
330       context.inRangeETag = context.request.getHeader(IF_RANGE_HEADER);
331       if(context.inRangeETag != null && validateStrong(context.inRangeETag, context.eTag ) ) {
332         if(!parseRanges(context)){
333           invalidRanges(context);
334           return true;
335         }
336       }
337     }
338     
339     if(context.inRangeDate == -1 && context.inRangeETag == null) {
340       if(!parseRanges(context)){
341         invalidRanges(context);
342         return true;
343       }
344     }
345     
346     context.ifNoneMatch = context.request.getHeader(IF_NONE_MATCH_HEADER);
347     if( context.ifNoneMatch != null && validateStrong(context.ifNoneMatch, context.eTag)) {
348       context.response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
349       context.response.setHeader(ETAG_HEADER, context.eTag);
350       context.response.setDateHeader(LAST_MODIFIED_HEADER, lastUpdated);
351       return true;
352     }
353     
354     context.ifModifiedSince = context.request.getDateHeader(IF_MODIFIED_SINCE_HEADER);
355     if( context.ifModifiedSince != -1 && context.ifModifiedSince >= lastUpdated ) {
356       context.response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
357       context.response.setHeader(ETAG_HEADER, context.eTag);
358       context.response.setDateHeader(LAST_MODIFIED_HEADER, lastUpdated);
359       return true;
360     }
361     
362     context.ifUnmodifiedSince = context.request.getDateHeader(IF_UNMODIFIED_SINCE_HEADER);
363     if( context.ifUnmodifiedSince != -1 && context.ifUnmodifiedSince < lastUpdated ) {
364       context.response.sendError(HttpServletResponse.SC_PRECONDITION_FAILED);
365       return true;
366     }
367     
368     return false;

397   /**
398    * Parses the Range Header of the request and sets the appropriate ranges list on the {@link FileRequestContext} Object.
399    * 
400    * @param context
401    * @return false if the range pattern contains no satisfiable ranges.
402    */
403   public static boolean parseRanges(FileRequestContext context) {
404     if( !StringUtils.isEmptyOrNull(context.range) ) {
405       Matcher rangeMatcher = rangePattern.matcher(context.range);
406       if(rangeMatcher.matches()) {
407         String ranges[] = rangeMatcher.group(1).split(",");
408         for(String range : ranges) {
409           long startBound = -1;
410           int hyphenIndex = range.indexOf('-');
411           if( hyphenIndex > 0 ) {
412             startBound = Long.parseLong(range.substring(0, hyphenIndex));
413           }
414           long endBound = -1;
415           if( hyphenIndex >= 0 && (hyphenIndex + 1) < range.length() ) {
416             endBound = Long.parseLong(range.substring(hyphenIndex + 1));
417           }
418           Range newRange = new Range(startBound, endBound);
419           
420           if(!(startBound != -1 && endBound != -1 && startBound > endBound) && !(startBound == -1 && endBound == -1)) {
421             context.ranges.add(newRange);
422           }
423         } 
424         return !context.ranges.isEmpty();
425       }
426     }
427     return true;
428   }

430   /**
431    * Parses an Accept header value and checks to see if the type is acceptable.
432    * @param headerValue The value of the header.
433    * @param strict Forces the type to be in the header.
434    * @param type The token that we need in order to be acceptable.
435    * @return
436    */
437   public static boolean canAccept(String headerValue, boolean strict, String type) {
438     if(headerValue != null && type != null)  {
439       String availableTypes[] = headerValue.split(",");
440       for(String availableType : availableTypes) {
441         String typeParams[] = availableType.split(";");
442         double qValue = 1.0d;
443         if(typeParams.length > 0) {
444           for(int i=1; i<typeParams.length; i++) {
445             if(typeParams[i].trim().startsWith("q=")) {
446               String qString = typeParams[i].substring(2).trim();
447               if(qString.matches("\\A\\d+(\\.\\d*){0,1}\\Z")){
448                 qValue = Double.parseDouble(qString);
449                 break;
450               } 
451             }
452           }
453         }
454         boolean matches = false;
455         if(typeParams[0].equals("*") || typeParams[0].equals("*/*")) {
456           matches = true;
457         } else {
458           matches = hasMatch(typeParams, type);
459         }
460         if(qValue == 0 && matches) {
461           return false;
462         } else {
463           return true;
464         }
465       }
466     }
467     return !strict;
468   }

470   /**
471    * Check to see if a Accept header accept part matches any of the given types.
472    * @param typeParams
473    * @param type
474    * @return
475    */
476   private static boolean hasMatch(String[] typeParams, String... type) {
477     boolean matches = false;
478     for(String t : type) {
479       for(String typeParam : typeParams) {
480         if(typeParam.contains("/")) {
481           String typePart = typeParam.replace("*", "");
482           if(t.startsWith(typePart) || t.endsWith(typePart)) {
483             matches = true;
484             break;
485           }
486         } else if(t.equals(typeParam)) {
487           matches = true;
488           break;
489         }
490       }
491       if(matches) {
492         break;
493       }
494     }
495     return matches;
496   }

507   public static List<String> parseETagList( String value ) {
508     List<String> etags = new ArrayList<String>();
509     value = value.trim();
510     if( "*".equals(value) ) {
511       etags.add(value);
512     }
513     else {
514       Matcher etagMatcher = etagPattern.matcher(value);
515       while( etagMatcher.lookingAt() ) {
516         etags.add(unescapePattern.matcher(etagMatcher.group(2)).replaceAll("$1"));
517         etagMatcher.region(etagMatcher.start()+etagMatcher.group().length(), etagMatcher.end());
518       }
519     }
520     return etags;
521   }

523   public static boolean validateStrong( String condition, String eTag ) {
524     List<String> parsed = parseETagList(condition);
525     if( parsed.size() == 1 && "*".equals(parsed.get(0)) ) return true;
526     else return parsed.contains(eTag);
527   }

529   /**
530    * Forces requests for index files to not use the file name.
531    * 
532    * @param context
533    * @return true if the request was handled, false otherwise.
534    * @throws IOException 
535    */
536   public boolean handleWelcomeRedirect( FileRequestContext context ) throws IOException {
537     if( context.file.isFile() && context.file.getName().equals("index.html")) {
538       resolveContentType(context);
539       String location = context.path.replaceFirst("/index.html\\Z", "");
540       if( location.isEmpty() ) location = "/";
541       if( context.request.getQueryString() != null ) location = location + "?" + context.request.getQueryString();
542       sendPermanentRedirect(context, location);
543       return true;
544     }
545     return false;
546   }
