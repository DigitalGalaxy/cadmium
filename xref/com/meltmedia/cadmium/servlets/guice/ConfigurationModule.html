16  package com.meltmedia.cadmium.servlets.guice;

18  import java.util.Set;
19  
20  import org.reflections.Reflections;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import com.google.inject.AbstractModule;
25  import com.google.inject.multibindings.Multibinder;
26  import com.meltmedia.cadmium.core.CadmiumModule;
27  import com.meltmedia.cadmium.core.config.CadmiumConfig;
28  import com.meltmedia.cadmium.core.config.ConfigurationClass;
29  import com.meltmedia.cadmium.core.config.ConfigurationParser;
30  import com.meltmedia.cadmium.core.config.ConfigurationParserClass;
31  import com.meltmedia.cadmium.core.config.ConfigurationParserFactory;
32  import com.meltmedia.cadmium.core.config.ConfigurationParserProvider;
33  import com.meltmedia.cadmium.core.config.impl.YamlConfigurationParser;
34  
35  /**
36   * Binds all {@link ConfigurationParser} related classes in Guice.
37   * 
38   * @author John McEntire
39   *
40   */
41  @CadmiumModule
42  public class ConfigurationModule extends AbstractModule {
43    private final Logger log = LoggerFactory.getLogger(getClass());
44  
45    @SuppressWarnings("rawtypes")
46    @Override
47    protected void configure() {
48      Multibinder<Class> configurationClassBinder = Multibinder.newSetBinder(binder(), Class.class, ConfigurationClass.class);
49      Reflections reflections = new Reflections("com.meltmedia.cadmium");
50      Set<Class<?>> configClasses = reflections.getTypesAnnotatedWith(CadmiumConfig.class);
51      log.debug("Found {} configuration classes", configClasses.size());
52      for(Class<?> configClass : configClasses) {
53        configurationClassBinder.addBinding().toInstance(configClass);
54        log.debug("Adding configuration type {}", configClass.getName());
55      }
56      
57      bind(Class.class).annotatedWith(ConfigurationParserClass.class).toInstance(YamlConfigurationParser.class);
58      
59      bind(ConfigurationParser.class).toProvider(ConfigurationParserProvider.class);
60      
61      bind(ConfigurationParserFactory.class);
62    }
63  
64  }
