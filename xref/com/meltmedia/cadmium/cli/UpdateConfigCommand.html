17  
18  import java.io.IOException;
19  import java.util.List;
20  
21  import org.apache.http.client.ClientProtocolException;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import com.beust.jcommander.Parameter;
26  import com.beust.jcommander.Parameters;
27  import com.meltmedia.cadmium.core.FileSystemManager;
28  import com.meltmedia.cadmium.core.git.GitService;
29  import com.meltmedia.cadmium.status.Status;

31  /**
32   * Sends a raw update config command to a Cadmium site.
33   * 
34   * @author Brian Barr
35   *
36   */
37  @Parameters(commandDescription = "Instructs a site to update its configuration.", separators="=")
38  public class UpdateConfigCommand extends AbstractAuthorizedOnly implements CliCommand {
39  
40    private final Logger log = LoggerFactory.getLogger(getClass());
41    public static final String UPDATE_CONFIG_ENDPOINT = "/system/update/config";
42  
43    @Parameter(names={"--repo"}, description="A new git repository url to switch to.", required=false)
44    private String repo;
45  
46    @Parameter(names={"--branch", "-b"}, description="The branch that you are updating to", required=false)
47    private String branch;
48  
49    @Parameter(names={"--revision", "-r"}, description="The revision that you are updating to", required=false)
50    private String revision;
51  
52    @Parameter(description="<site>", required=true)
53    private List<String> site;
54  
55    @Parameter(names={"--message", "-m"}, description="The comment for the history", required=true)
56    private String message;
57  
58    @Parameter(names={"--force", "-f"}, description="Force the update", required=false)
59    private boolean force;
60  
61    public void execute() throws ClientProtocolException, IOException {
62  
63      String siteUrl = getSecureBaseUrl(site.get(0));
64  
65      System.out.println("Getting status of ["+ siteUrl +"]");
66      GitService gitValidation = null;
67      try {
68  
69        Status siteStatus = StatusCommand.getSiteStatus(siteUrl, token);
70  
71        boolean repoSame = false;
72        boolean branchSame = false;
73        boolean revisionSame = false;
74        boolean forceUpdate = force;      
75  
76        String currentConfigRepo = siteStatus.getConfigRepo();
77        String currentConfigRevision = siteStatus.getConfigRevision();
78        String currentConfigBranch = siteStatus.getConfigBranch();
79  
80  
81        log.debug("config repo = {}, and currentConfigRepo = {}", repo, currentConfigRepo);
82  
83        if(repo != null && repo.trim().equals(currentConfigRepo.trim())) {
84  
85          repoSame = true;
86        }
87  
88        log.debug("branch = {}, and currentBranch = {}", branch, currentConfigBranch);
89  
90        if(branch != null && branch.trim().equals(currentConfigBranch.trim())) {
91  
92          branchSame = true;
93        }
94  
95        log.debug("revision = {}, and currentRevision = {}", revision, currentConfigRevision);
96  
97        if(revision != null && revision.trim().equals(currentConfigRevision.trim())) {
98  
99          revisionSame = true;
100       }
101 
102       log.debug("branchSame = {}, and revisionSame = {}", branchSame, revisionSame);
103 
104       if(repoSame && branchSame && revisionSame && !forceUpdate) {
105 
106         System.out.println("The config for site [" + siteUrl  + "] is already on repo [" + repo + "] branch [" + branch + "] and revision [" + revision  + "].");
107       }
108       else {    
109         if(repo != null) {
110           siteStatus.setRepo(repo);
111           siteStatus.setBranch(null);
112           siteStatus.setRevision(null);
113         }
114         if(repo == null) {
115           repo = siteStatus.getRepo();
116         }
117         gitValidation = CloneCommand.cloneSiteRepo(siteStatus);
118         String newBranch = null;
119         String rev = null;
120 
121         if(branch == null || branch.length() == 0) {
122           gitValidation.switchBranch(siteStatus.getConfigBranch());
123         }
124         if(branch != null) {
125           if(gitValidation.isBranch(branch)) {
126             newBranch = branch;
127             gitValidation.switchBranch(branch);
128             log.debug("branch being added = {}", branch);
129           } else {
130             System.err.println("The branch ["+branch+"] does not exist.");
131             throw new Exception("");
132           }
133         }
134 
135         if(revision != null) {
136           if(gitValidation.checkRevision(revision)){
137             rev = revision;
138             log.debug("revision being added = {}", revision);
139           } else {  
140             System.err.println("Revision ["+revision+"] does not exist on the branch ["+gitValidation.getBranchName()+"]");
141             throw new Exception("");
142           }
143         } 
144 
145         if(UpdateCommand.sendUpdateMessage(siteUrl, repo, newBranch, rev, message, token, UPDATE_CONFIG_ENDPOINT)){
146           System.out.println("Update successful");
147         }
148 
149       }
150 
151     } 
152     catch (Exception e) {
153 
154       System.err.println("Failed to updated site [" + siteUrl  + "] to repo [" + repo + "] branch [" + branch  + "] and revision [" + revision  + "].");
155     } finally {
156       if(gitValidation != null) {
157         try {
158           FileSystemManager.deleteDeep(gitValidation.getBaseDirectory());
159         } catch (Exception e) {
160         }
161       }
162     }
163 
164   }
165 
166   @Override
167   public String getCommandName() {
168     return "update-config";
169   }
