36  /**
37   * Implements functionality to facilitate authentication to Cadmium rest endpoints.
38   * 
39   * @author John McEntire
40   * @author Christian Trimble
41   *
42   */
43  public class AbstractAuthorizedOnly implements AuthorizedOnly {
44    private static final Logger logger = LoggerFactory.getLogger(AbstractAuthorizedOnly.class);
45    
46    private static final Pattern URL_PATTERN = compilePattern("\\Ahttp://([^/:]+)(?:\\:\\d+)?(?:/[^/]*)*\\Z", logger);
47  
48    /**
49     * Compiles the given regex.  Returns null if the pattern could not be compiled and logs an error message to the
50     * specified logger.
51     *
52     * @param regex the regular expression to compile.
53     * @param logger the logger to notify of errors.
54     * @return The compiled regular expression, or null if it could not be compiled.
55     */
56    private static Pattern compilePattern( String regex, Logger logger ) {
57      try {
58        return Pattern.compile(regex);
59      }
60      catch( PatternSyntaxException pse ) {
61        logger.error("Could not compile regex "+regex, pse);
62      }
63      return null;
64    }

69    protected String token;

74    @Override

76      this.token = token;

88     * Adds the Authorization Header to the given message.
89     * @param token The users Github API Token.

92    protected static void addAuthHeader(String token, HttpMessage message) {
93      message.addHeader("Authorization", "token " + token);
94    }

102     addAuthHeader(token, message);
103   }

105   /**
106    * Converts the passed in siteUrl to be https if not already or not local.
107    * 
108    * @param siteUrl The url of a cadmium site.
109    * @return The passed in siteUrl or the same url but converted to https.
110    */
111   protected String getSecureBaseUrl(String siteUrl) {
112     Matcher urlMatcher = URL_PATTERN.matcher(siteUrl);
113     if(urlMatcher.matches()) {
114       logger.debug("Url matches [{}]", siteUrl);
115       boolean local = false;
116       try {
117         logger.debug("Checking if host [{}] is local", urlMatcher.group(1));
118         InetAddress hostAddr = InetAddress.getByName(urlMatcher.group(1));
119         local = hostAddr.isLoopbackAddress() || hostAddr.isSiteLocalAddress();
120         logger.debug("IpAddress [{}] local: {}", hostAddr.getHostAddress(), local);
121       } catch(UnknownHostException e) {
122         logger.warn("Hostname not found ["+siteUrl+"]", e);
123       }
124       if(!local) {
125         return siteUrl.replaceFirst("http://", "https://");
126       }
127     }
128     return siteUrl;
129   }

141   /**
142    * Sets the Commons HttpComponents to accept all SSL Certificates.
143    * 
144    * @throws Exception
145    * @return The reference passed in.
146    * @throws KeyStoreException 
147    * @throws NoSuchAlgorithmException 
148    * @throws UnrecoverableKeyException 
149    * @throws KeyManagementException 
150    */
151   protected static DefaultHttpClient setTrustAllSSLCerts(DefaultHttpClient client) throws KeyManagementException, UnrecoverableKeyException, NoSuchAlgorithmException, KeyStoreException {
152     SSLSocketFactory acceptAll = new SSLSocketFactory(new TrustSelfSignedStrategy(), SSLSocketFactory.ALLOW_ALL_HOSTNAME_VERIFIER);
153     
154     client.getConnectionManager().getSchemeRegistry().register(new Scheme("https", 443, acceptAll));
155     
156     return client;
157   }
