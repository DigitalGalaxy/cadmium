16  package com.meltmedia.cadmium.core.worker;
17  
18  import java.io.File;
19  import java.util.concurrent.Callable;
20  import java.util.concurrent.ExecutorService;
21  import java.util.concurrent.Future;
22  import java.util.concurrent.RejectedExecutionException;
23  
24  import javax.inject.Inject;
25  import javax.inject.Named;
26  
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import com.meltmedia.cadmium.core.ContentGitService;
31  import com.meltmedia.cadmium.core.SiteDownService;
32  import com.meltmedia.cadmium.core.git.DelayedGitServiceInitializer;
33  import com.meltmedia.cadmium.core.git.GitService;
34  import com.meltmedia.cadmium.core.meta.SiteConfigProcessor;
35  
36  public class CheckInitializedTask implements Callable<Boolean> {
37    private final Logger logger = LoggerFactory.getLogger(getClass());
38    private Callable<GitService> taskToCheck = null;
39    private ExecutorService pool = null;
40    private Future<GitService> futureTask = null;
41    private SiteDownService service = null;
42    private SiteConfigProcessor metaProcessor = null;
43    private DelayedGitServiceInitializer gitInit = null;
44    private String sharedContentRoot = null;
45    private String warName = null;
46    
47    @Inject
48    public CheckInitializedTask(InitializeTask taskToCheck, SiteConfigProcessor metaProcessor, SiteDownService service, @ContentGitService DelayedGitServiceInitializer gitInit, @Named("sharedContentRoot") String contentRoot, @Named("warName") String warName) {
49      this.taskToCheck = taskToCheck;
50      this.metaProcessor = metaProcessor;
51      this.service = service;
52      this.gitInit = gitInit;
53      this.sharedContentRoot = contentRoot;
54      this.warName = warName;
55    }
56    
57    public CheckInitializedTask setFuture(Future<GitService> futureTask) {
58      this.futureTask = futureTask;
59      return this;
60    }
61    
62    public CheckInitializedTask setExecutor(ExecutorService pool) {
63      this.pool = pool;
64      return this;
65    }
66  
67    @Override
68    public Boolean call() throws Exception {
69      GitService git = null;
70      try {
71        if((git = futureTask.get()) == null) {
72          File gitCheckout = new File(new File(sharedContentRoot, warName), "git-checkout");
73          if(gitCheckout.exists() && gitCheckout.isDirectory()) {
74            try {
75              git = GitService.createGitService(gitCheckout.getAbsolutePath());
76              gitInit.setGitService(git);
77            } catch(Exception e){
78              logger.warn("Git Service did not initialize.", e);
79            }
80          }
81          return true;
82        }
83      } catch(Exception e){
84        try {
85          logger.info("Initialize task `" + taskToCheck + "` failed.  Retrying in 30 seconds.", e);
86          Thread.sleep(30000l);
87        } catch(Exception e1){}
88        try {
89          futureTask = pool.submit(taskToCheck);
90          pool.submit(this);
91        } catch(RejectedExecutionException e1){
92          logger.info("ExecutorService has been shutdown.", e1);
93        }
94        return true;
95      }
96      if(metaProcessor != null) {
97        logger.info("Making meta config live `{}`", taskToCheck);
98        metaProcessor.makeLive();
99      }
100     logger.info("Stopping Maintenance page `{}`", taskToCheck);
101     service.stop();
102     gitInit.setGitService(git);
103     return true;
104   }
