197   public static HttpServletResponse mockResponse(final Map<String, List<String>> headers, OutputStream output) throws IOException {
198     HttpServletResponse response = mock(HttpServletResponse.class);
199     when(response.getOutputStream()).thenReturn(new MockServletOutputStream(output));
200     if( headers != null ) {
201     Answer<Void> addHeaderAnswer = new Answer<Void>() {
202       @Override
203       public Void answer(InvocationOnMock invocation) throws Throwable {
204         Object[] args = invocation.getArguments();
205         List<String> value = headers.get((String)args[0]);
206         if( value == null ) headers.put((String)args[0], value = new ArrayList<String>());
207         value.add(args[1].toString());
208         return null;
209       }
210     };
211     Answer<Void> setHeaderAnswer = new Answer<Void>() {
212       @Override
213       public Void answer(InvocationOnMock invocation) throws Throwable {
214         Object[] args = invocation.getArguments();
215         List<String> value = headers.get((String)args[0]);
216         if( value == null ) headers.put((String)args[0], value = new ArrayList<String>());
217         value.clear();
218         value.add(args[1].toString());
219         return null;
220       }
221     };
222     doAnswer(setHeaderAnswer).when(response).setHeader(anyString(), anyString());
223     doAnswer(setHeaderAnswer).when(response).setDateHeader(anyString(), anyLong());
224     doAnswer(setHeaderAnswer).when(response).setIntHeader(anyString(), anyInt());
225     doAnswer(addHeaderAnswer).when(response).addHeader(anyString(), anyString());
226     doAnswer(addHeaderAnswer).when(response).addDateHeader(anyString(), anyLong());
227     doAnswer(addHeaderAnswer).when(response).addIntHeader(anyString(), anyInt());
228     }
229     return response;
230   }

268   public static String rangeHeader(RangeSpec... rangeSpecs) {
269     StringBuilder rangeBuilder = new StringBuilder().append("bytes=");
270    for( int i = 0; i < rangeSpecs.length; i++ ) {
271       rangeBuilder.append(rangeSpecs[i].start).append("-").append((rangeSpecs[i].end-1));
272       if( i < rangeSpecs.length - 1) rangeBuilder.append(",");
273     }
274    return rangeBuilder.toString();
275   }

337   public static ResponseVerifier success(final String name, final String etag, final Long lastModified, final boolean shouldGzip, final ResponseVerifier... ranges) {
338     return new ResponseVerifier() {
339       @Override
340       public void verifyResponse(HttpServletResponse response) throws IOException, ServletException {
341         verify(response).setHeader(CONTENT_DISPOSITION_HEADER, "inline;filename=\""+name+"\"");
342         verify(response).setHeader(ACCEPT_RANGES_HEADER, "bytes");
343         verify(response).setHeader(ETAG_HEADER, etag);
344         verify(response).setDateHeader(LAST_MODIFIED_HEADER, lastModified);
345         if( ranges.length == 0 ) {
346           verify(response).setStatus(200);
347           verify(response).setHeader(eq(CONTENT_RANGE_HEADER), anyString());
348           verify(response).setContentType(TEXT_HTML_TYPE+";charset=UTF-8");
349           if(shouldGzip) {
350             verify(response).setHeader(CONTENT_ENCODING_HEADER, "gzip");
351           } else {
352             verify(response).setHeader(eq(CONTENT_LENGTH_HEADER), anyString());
353           }
354         }
355         else {
356           for( ResponseVerifier range : ranges ) {
357             range.verifyResponse(response);
358           }
359         }
360       } 
361     };
362   }
