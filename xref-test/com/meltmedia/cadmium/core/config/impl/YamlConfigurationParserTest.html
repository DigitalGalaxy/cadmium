36    @Test
37    public void testParseDirectory() throws Exception {
38      YamlConfigurationParser parser = new YamlConfigurationParser();
39      parser.setConfigurationClasses(Arrays.asList(new Class[] {TestConfigPojo.class}));
40      parser.setEnvironment("production");
41      parser.parseDirectory(new File("src/test/resources/test-configurations"));
42      
43      assertTrue("No configuration was parsed.", !parser.configuration.isEmpty());
44      assertTrue("No default env was found.", parser.configuration.containsKey(YamlConfigurationParser.DEFAULT));
45      Map<String, ?> defaultSection = parser.configuration.get(YamlConfigurationParser.DEFAULT);
46      assertTrue("Wrong number of configurations found in default env.", defaultSection.size() == 2);
47      validateConfig(defaultSection, "test", "test", "value", 15);
48      validateConfig(defaultSection, "test2", "test2", "another value", 17);
49      
50      assertTrue("No production env was found.", parser.configuration.containsKey("production"));
51      Map<String, ?> prodSection = parser.configuration.get("production");
52      assertTrue("Wrong number of configurations found in production env.", prodSection.size() == 7);
53      validateConfig(prodSection, "test", "updated value", "value", 15);
54      validateConfig(prodSection, "test2", "updated value 2", "another value", 17);
55      validateConfig(prodSection, "test3", "test 3", "yet another value", 99);
56      validateType(prodSection, "test4", Number.class);
57      validateType(prodSection, "test5", Collection.class);
58      validateType(prodSection, "test6", Map.class);
59      validateType(prodSection, "test7", Collection.class);
60      
61    }

63    @SuppressWarnings("unchecked")
64    @Test
65    public void testGetConfiguration() throws Exception {
66      YamlConfigurationParser parser = new YamlConfigurationParser();
67      parser.setConfigurationClasses(Arrays.asList(new Class[] {TestConfigPojo.class}));
68      parser.setEnvironment("production");
69      
70      parser.configuration.put(YamlConfigurationParser.DEFAULT, new HashMap<String, Object>());
71      parser.configuration.put("production", new HashMap<String, Object>());
72      
73      TestConfigPojo defaultTest = new TestConfigPojo();
74      defaultTest.setName("test");
75      defaultTest.setField("field");
76      defaultTest.setAnotherField(55);
77      ((Map<String, Object>)parser.configuration.get(YamlConfigurationParser.DEFAULT)).put("test", defaultTest);
78      
79      TestConfigPojo prodTest = new TestConfigPojo();
80      prodTest.setName("test2");
81      prodTest.setField("field2");
82      prodTest.setAnotherField(66);
83      ((Map<String, Object>)parser.configuration.get("production")).put("test", prodTest);
84      
85      assertTrue("Got wrong instance [production]", parser.getConfiguration("test", TestConfigPojo.class) == prodTest);
86      parser.environment = "development";
87      assertTrue("Got wrong instance [development]", parser.getConfiguration("test", TestConfigPojo.class) == defaultTest);
88      
89    }

91    private void validateConfig(Map<String, ?> defaultSection, String key, String name, String field, Integer anotherField) {
92      assertTrue("No "+key+" section was not found in section.", defaultSection.containsKey(key));
93      assertTrue(key+" section was not correct type.", defaultSection.get(key) instanceof TestConfigPojo);
94      TestConfigPojo config = (TestConfigPojo) defaultSection.get(key);
95      assertTrue("Incorrect name ["+config.getName()+"]", name.equals(config.getName()));
96      assertTrue("Incorrect field ["+config.getField()+"]", field.equals(config.getField()));
97      assertTrue("Incorrect anotherField ["+config.getAnotherField()+"]", anotherField.equals(config.getAnotherField()));
98    }
99    
100   private void validateType(Map<String, ?> section, String key, Class<?> type) {
101     assertTrue("No "+key+" section was not found in section.", section.containsKey(key));
102     Object value = section.get(key);
103     assertTrue("Value is not expected type. "+type, type.isAssignableFrom(value.getClass()));
104     System.out.println(key+"=["+value+"]");
105   }
