18  import static org.junit.Assert.assertTrue;

20  import org.junit.Test;

22  public class RedirectConfigProcessorTest {
23  
24    @Test
25    public void requestMatchesTest() throws Exception {
26      RedirectConfigProcessor proc = new RedirectConfigProcessor();
27      proc.liveRedirects.add(new Redirect());
28      proc.liveRedirects.get(0).setPath("path1");
29      proc.liveRedirects.get(0).setUrl("url");
30      proc.liveRedirects.add(new Redirect());
31      proc.liveRedirects.get(1).setPath("path2\\?info");
32      proc.liveRedirects.get(1).setUrl("url2");
33      
34      Redirect redir = proc.requestMatches("path1", null);
35      assertTrue("Path1 should have matched", redir != null && redir.getUrlSubstituted().equals("url"));
36      redir = proc.requestMatches("path1", "info2");
37      assertTrue("Path1 with queryString should have matched", redir != null && redir.getUrl().equals("url"));
38      redir = proc.requestMatches("path2", "info");
39      assertTrue("Path2 with query should have matched", redir != null && redir.getUrlSubstituted().equals("url2"));
40      assertTrue("Path2 should not have matched", proc.requestMatches("path2", null) == null);
41      assertTrue("Path3 should not have matched", proc.requestMatches("path3", null) == null);
42    }
43    
44    @Test
45    public void makeLiveTest() throws Exception {
46      RedirectConfigProcessor proc = new RedirectConfigProcessor();
47      proc.stagedRedirects.add(new Redirect());
48      proc.stagedRedirects.get(0).setPath("path");
49      proc.stagedRedirects.get(0).setUrl("url");
50      
51      proc.makeLive();
52      
53      assertTrue("Redirects not promoted", proc.liveRedirects.size() == 1);
54      assertTrue("Redirects not promoted correctly", proc.liveRedirects.get(0).getPath().equals("path") && proc.liveRedirects.get(0).getUrl().equals("url"));
55    }

57    @Test
58    public void processFromDirectoryTest() throws Exception {
59      RedirectConfigProcessor proc = new RedirectConfigProcessor();
60      proc.processFromDirectory("./src/test/resources");
61      
62      assertTrue("Redirects not parsed.", proc.stagedRedirects.size() == 3);
63      for(Redirect redir : proc.stagedRedirects) {
64        assertTrue("Redirect not parsed", redir.getUrl() != null && redir.getUrl().length() > 0 && redir.getPath() != null && redir.getPath().length() > 0);
65      }
66    }
