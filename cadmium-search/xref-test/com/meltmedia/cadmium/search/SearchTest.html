18  import static org.junit.Assert.assertTrue;

23  import org.junit.After;
24  import org.junit.Before;
25  import org.junit.Test;

27  import com.meltmedia.cadmium.core.FileSystemManager;

29  public class SearchTest {
30  
31    private SearchContentPreprocessor preprocessor;
32    private SearchService service;
33    
34    @Before
35    public void setupIndexes() throws Exception {
36      preprocessor = new SearchContentPreprocessor();
37      preprocessor.processFromDirectory("./target/test-classes/test-content/META-INF");
38      service = new SearchService();
39      service.setIndexSearchProvider(preprocessor);
40      
41      preprocessor.makeLive();
42      
43    }
44    
45    @SuppressWarnings("unchecked")
46    @Test
47    public void testSearchNoResults() throws Exception {
48      Map<String, Object> results = service.search("bad_query");
49      
50      assertTrue("Should not be empty", !results.isEmpty());
51      assertTrue("Should be set", results.containsKey("number-hits"));
52      assertTrue("Should be integer", results.get("number-hits") instanceof Integer);
53      assertTrue("Should say 0", new Integer(0).equals(results.get("number-hits")));
54      assertTrue("Should have results entry", results.containsKey("results"));
55      assertTrue("Should have no result entries", ((List<Map<String, Object>>)results.get("results")).isEmpty());
56    }
57    
58    @After
59    public void finishUp() throws Exception {
60      preprocessor.finalize();
61      FileSystemManager.deleteDeep(preprocessor.getIndexDir().getAbsolutePath());
62    }
