28  /**
29   * <p>CLI Command that clones the content from a source to a target.</p>
30   * <p>This simply checks the status of both source and target sites. 
31   * Then, as long as they are both using the same Git repository, 
32   * issues an update command on the target site with the branch and revision of the source site.</p>
33   * <p>The @Parameters of this class get wired up by JCommander.</p>
34   * 
35   * @see <a href="http://jcommander.org/">JCommander</a>
36   * 
37   * @author John McEntire
38   *
39   */
40  @Parameters(commandDescription="This command will move all content from one branch to another or tag a version of a branch.", separators="=")
41  public class CloneCommand extends AbstractAuthorizedOnly implements CliCommand {
42  
43    /**
44     * A collection of strings that will be assigned the raw parameters that are not associated with an option.
45     * 
46     * @see <a href="http://jcommander.org/">JCommander</a>
47     */
48    @Parameter(description="<source-site> <target-site>", required=true)
49    private List<String> sites;
50    
51    /**
52     * The command to log in the update history of the target site.
53     * 
54     * @see <a href="http://jcommander.org/">JCommander</a>
55     */
56    @Parameter(names={"--message", "-m"}, description="comment", required=true)
57    private String comment;
58    
59    /**
60     * 
61     * @see <a href="http://jcommander.org/">JCommander</a>
62     */
63    @Parameter(names={"--include-config", "-c"}, description="Clone configuration", required=false)
64    private boolean includeConfig = false;
65    
66    /**
67     * Called to execute this command.
68     */
69    public void execute() throws Exception {
70      String site1 = getSecureBaseUrl(sites.get(0));
71      String site2 = getSecureBaseUrl(sites.get(1));
72      if(site1.equalsIgnoreCase(site2)) {
73        System.err.println("Cannot clone a site into itself.");
74        System.exit(1);
75      }
76      
77      try{
78        System.out.println("Getting status of ["+site1+"]");
79        Status site1Status = StatusCommand.getSiteStatus(site1, token);
80        
81        System.out.println("Getting status of ["+site2+"]");
82        Status site2Status = StatusCommand.getSiteStatus(site2, token);
83        if(site1Status != null && site2Status != null) {
84          
85          String repo = site1Status.getRepo();
86          String revision = site1Status.getRevision();
87          String branch = site1Status.getBranch();
88          
89          if(site2Status.getRepo().equals(repo) && site2Status.getBranch().equals(branch) && site2Status.getRevision().equals(revision)) {
90            System.err.println("Source [" + site1 + "] is on the same content repo, branch, and revision as the target [" + site2 + "].");
91            checkSendUpdateConfigMessage(site1, site2, site1Status, site2Status);
92            System.exit(1);
93          }
94          
95          System.out.println("Sending update message to ["+site2+"]");
96          UpdateCommand.sendUpdateMessage(site2, repo, branch, revision, "Cloned from ["+site1+"]: " + comment, token);
97          checkSendUpdateConfigMessage(site1, site2, site1Status, site2Status);
98        } else {
99          System.err.println("Failed to get status from source and/or target.");
100         System.exit(1);
101       }
102     } catch(Exception e) {
103       e.printStackTrace();
104       System.err.println("Failed to clone ["+site1+"] to ["+site2+"]: "+e.getMessage());
105     }
106   }

108   /**
109    * Checks if a config update message is necessary and sends it if it is.
110    * 
111    * @param site1 The source site for the configuration.
112    * @param site2 The target site for the configuration.
113    * @param site1Status The status of the source site.
114    * @param site2Status The status of the target site.
115    * @throws Exception
116    */
117   private void checkSendUpdateConfigMessage(String site1, String site2,
118       Status site1Status, Status site2Status) throws Exception {
119     String repo;
120     String revision;
121     String branch;
122     if(includeConfig) {
123       repo = site1Status.getConfigRepo();
124       revision = site1Status.getConfigRevision();
125       branch = site1Status.getConfigBranch();
126       if(!StringUtils.isEmptyOrNull(repo) && !StringUtils.isEmptyOrNull(revision) && !StringUtils.isEmptyOrNull(branch)) {
127         if(!repo.equals(site2Status.getConfigRepo()) || !revision.equals(site2Status.getConfigRevision()) || ! branch.equals(site2Status.getConfigBranch())) {
128           System.out.println("Sending update/config message to ["+site2+"]");
129           UpdateCommand.sendUpdateMessage(site2, repo, branch, revision, "Cloned config from ["+site1+"]: " + comment, token, UpdateConfigCommand.UPDATE_CONFIG_ENDPOINT);
130         } else {
131           System.out.println("Source [" + site1 + "] is on the same configuration repo, branch, and revision as the target [" + site2 + "].");
132         }
133       } else {
134         System.out.println("Configuration status not available from source site [" + site1 + "]");
135       }
136     }
137   }

139   /**
140    * Replaces the content contained within the Git repository branch pointed to by the {@link GitService} object with the source.
141    * 
142    * @param source The directory to commit the content from.
143    * @param service The {@link GitService} pointing to the repository and branch to commit to.
144    * @param comment The commit message.
145    * @return The new revision.
146    * @throws Exception
147    */
148   public static String cloneContent(String source, GitService service, String comment) throws Exception {
149     String rev = GitService.moveContentToBranch(source, service, service.getBranchName(), comment);
150     service.push(false);
151     return rev;
152   }

154   /**
155    * Clones a repository locally from a status response from a Cadmium site.
156    * @param status The status response to clone locally.
157    * @return The {@link GitService} object that points to the newly cloned remote repo.
158    * @throws Exception
159    */
160   public static GitService cloneSiteRepo(Status status) throws Exception {
161     File tmpDir = File.createTempFile("site", "git");
162     GitService git = null;
163     if(tmpDir.delete()) {
164       try {
165         git = GitService.cloneRepo(status.getRepo(), tmpDir.getAbsolutePath());
166         if(status.getBranch() != null && !git.getBranchName().equals(status.getBranch())) {
167           git.switchBranch(status.getBranch());
168         }
169         if(status.getRevision() != null && !git.getCurrentRevision().equals(status.getRevision())) {
170           git.resetToRev(status.getRevision());
171         }
172       } catch(Exception e) {
173         System.err.println("Failed to clone repo "+status.getRepo()+" branch "+status.getBranch()+ "["+tmpDir+"]");
174         e.printStackTrace();
175         if(git != null) {
176           git.close();
177           git = null;
178         }
179       }
180     }
181     return git;
182   }
