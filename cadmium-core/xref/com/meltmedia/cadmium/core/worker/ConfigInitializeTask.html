16  package com.meltmedia.cadmium.core.worker;
17  
18  import java.io.File;
19  import java.util.Properties;
20  import java.util.concurrent.Callable;
21  
22  import javax.inject.Inject;
23  import javax.inject.Named;
24  
25  import org.eclipse.jgit.api.errors.RefNotFoundException;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import com.meltmedia.cadmium.core.config.ConfigManager;
30  import com.meltmedia.cadmium.core.git.GitService;
31  import com.meltmedia.cadmium.core.history.HistoryManager;
32  
33  public class ConfigInitializeTask implements Callable<GitService> {
34    private final Logger logger = LoggerFactory.getLogger(getClass());
35    
36    private String contentRoot = null;
37    private String warName = null;
38    private ConfigManager configManager;  
39    private HistoryManager historyManager = null;
40  
41    @Inject
42    public ConfigInitializeTask(ConfigManager configManager, @Named("sharedContentRoot") String contentRoot, @Named("warName") String warName, HistoryManager historyManager) {
43      
44      this.contentRoot = contentRoot;
45      this.warName = warName;
46      this.configManager = configManager;
47      this.historyManager = historyManager;
48    }
49  
50    @Override
51    public GitService call() throws Exception {
52      
53      Properties configProperties = configManager.getDefaultProperties();
54      String branch = configProperties.getProperty("com.meltmedia.cadmium.config.branch");
55      String repoUri = configProperties.getProperty("com.meltmedia.cadmium.config.git.uri", configProperties.getProperty("com.meltmedia.cadmium.git.uri"));
56      
57      GitService cloned = null;
58      if(repoUri != null && branch != null) {
59        Throwable t = null;
60  
61        try {
62          logger.info("Attempting to initialize config for `{}` into `{}`", warName, contentRoot);
63          cloned = GitService.initializeConfigDirectory(repoUri, branch, contentRoot, warName, historyManager, configManager);
64  
65          String contentDirectory = configProperties.getProperty("com.meltmedia.cadmium.config.lastUpdated");
66          
67          if(configManager != null && contentDirectory != null) {
68            logger.info("Processing configuration dir in `{}`", warName);
69            this.configManager.parseConfigurationDirectory(new File(contentDirectory));
70          }
71          configManager.makeConfigParserLive();
72          logger.info("Successfully initialized config for `{}`", warName);
73        } catch(RefNotFoundException e) {
74          logger.warn("Branch `"+branch+"` does not exist.", e);
75          return null;
76        } catch(Throwable t1) {
77          logger.error("Failed to initialize config git repo `"+warName+"`", t);
78          t = t1;
79        }
80        if(t != null) {
81          throw new Exception(t);
82        }
83      } else {
84        logger.warn("Invalid setting for configuration repo {} and branch {}", repoUri, branch);
85      }
86      return cloned;
87    }
88  
89    @Override
90    public String toString() {
91      return warName;
92    }
93  
94  }
