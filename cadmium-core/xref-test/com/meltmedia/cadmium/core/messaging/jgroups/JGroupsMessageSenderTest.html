32  public class JGroupsMessageSenderTest {
33    private static final String serializedMessage = "{\"command\":\"UPDATE\",\"protocolParameters\":{\"branch\":\"master\",\"rev\":\"HEAD\"},\"requestTime\":233434800}";
34    private static final Message deserializedMessage = new Message();
35    static {
36      deserializedMessage.setCommand(ProtocolMessage.UPDATE);
37      deserializedMessage.setRequestTime(new Long(233434800));
38      deserializedMessage.setProtocolParameters(new LinkedHashMap<String, String>());
39      deserializedMessage.getProtocolParameters().put("branch", "master");
40      deserializedMessage.getProtocolParameters().put("rev", "HEAD");
41    }
42  
43    @Test
44    public void testSendMessage() throws Exception {
45      Address me = new IpAddress(12345);
46      Vector<Address> viewMems = new Vector<Address>();
47      viewMems.add(me);
48      
49      DummyJChannel channel = new DummyJChannel(me, viewMems);
50      
51      JGroupsMessageSender sender = new JGroupsMessageSender();
52      sender.channel = channel;
53      
54      sender.sendMessage(deserializedMessage, new ChannelMember(me));
55      
56      assertTrue("Failed to send message", channel.getMessageList().size() == 1 && channel.getMessageList().get(0) != null);
57      assertEquals("Incorrect Destination", me.toString(), channel.getMessageList().get(0).getDest().toString());
58      assertEquals("Incorrect Message sent", serializedMessage, channel.getMessageList().get(0).getObject().toString());
59    }
60  }
