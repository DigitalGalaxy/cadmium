52      MaintenanceCommandAction maintCmd = new MaintenanceCommandAction();

54      maintCmd.manager = new HistoryManager(null);
55      CommandContext ctx = new CommandContext(new IpAddress(1234), new Message());
56      ctx.getMessage().setCommand(ProtocolMessage.MAINTENANCE);
57      ctx.getMessage().getProtocolParameters().put("state", "on");
58      ctx.getMessage().getProtocolParameters().put("comment", "comment");
59      
60      assertTrue("Command failed", maintCmd.execute(ctx));
61      assertTrue("Site Down is On", siteDownService.isOn());
62      
63      ctx.getMessage().getProtocolParameters().put("state", "off");
64      assertTrue("Command failed", maintCmd.execute(ctx));
65      Assert.assertEquals("Site Down is Off", false, siteDownService.isOn());
