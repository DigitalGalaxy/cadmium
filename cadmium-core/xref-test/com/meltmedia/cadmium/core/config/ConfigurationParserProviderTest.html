16  package com.meltmedia.cadmium.core.config;

18  import static org.junit.Assert.*;
19  
20  import java.io.File;
21  import java.util.Collection;
22  import java.util.HashSet;

24  import org.junit.Before;
25  import org.junit.BeforeClass;
26  import org.junit.Test;
27  
28  public class ConfigurationParserProviderTest {
29    
30    private ConfigurationParserProvider provider;
31    
32    @BeforeClass
33    public static void setupSystem() {
34      System.setProperty("com.meltmedia.cadmium.environment", "testing");
35    }
36    
37    @SuppressWarnings("rawtypes")
38    @Before
39    public void testProviderConstructor() throws Exception {
40      provider = new ConfigurationParserProvider(ConfigurationParserImpl.class);
41      provider.configurationClasses = new HashSet<Class>();
42      provider.configurationClasses.add(DummyClass.class);
43    }
44    
45    @Test
46    public void testProviderGet() throws Exception {
47      ConfigurationParser parser1 = provider.get();
48      validateParser(parser1);
49      ConfigurationParser parser2 = provider.get();
50      validateParser(parser2);
51      assertTrue("Provider provided exact same instance for 2 calls to get.", parser1 != parser2);
52    }
53    
54    public void validateParser(ConfigurationParser parser) {
55      assertNotNull("Provider provided null.", parser);
56      assertTrue("Parser is not of the expected type.", parser instanceof ConfigurationParserImpl);
57      ConfigurationParserImpl impl = (ConfigurationParserImpl) parser;
58      assertNotNull("Configuration classes not wired in.", impl.configurationClasses);
59      assertEquals("Parser passed a different instance of the configuration classes.", impl.configurationClasses, provider.configurationClasses);
60      assertNotNull("Environment not wired in.", impl.environment);
61      assertEquals("Wrong environment wired in.", impl.environment, "testing");
62    }
63    
64    
65    public static class DummyClass {}
66    
67    public static class ConfigurationParserImpl implements ConfigurationParser {
68      
69      @SuppressWarnings("rawtypes")
70      Collection<Class> configurationClasses;
71      String environment;
72  
73      @SuppressWarnings("rawtypes")
74      @Override
75      public void setConfigurationClasses(
76          Collection<Class> configurationClasses) {
77        this.configurationClasses = configurationClasses;
78      }
79  
80      @Override
81      public void setEnvironment(String environment) {
82        this.environment = environment;
83      }
84  
85      @Override
86      public void parseDirectory(File configurationDirectory) throws Exception {
87        
88      }
89  
90      @Override
91      public <T> T getConfiguration(String key, Class<T> type)
92          throws ConfigurationNotFoundException {
93        return null;
94      }
95      
96    }
